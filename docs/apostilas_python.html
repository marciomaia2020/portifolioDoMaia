Composição e Herança:

Utilize composição para reutilizar código sem criar uma hierarquia rígida.


class Motor:
    def ligar(self):
        print("Motor ligado")

class Carro:
    def __init__(self, motor):
        self.motor = motor

    def ligar_carro(self):
        self.motor.ligar()

motor = Motor()
carro = Carro(motor)
carro.ligar_carro()  # Saída: Motor ligado

Decorators:


def meu_decorator(func):
    def wrapper():
        print("Algo antes da função")
        func()
        print("Algo depois da função")
    return wrapper

@meu_decorator
def diz_oi():
    print("Oi!")

diz_oi()
# Saída:
# Algo antes da função
# Oi!
# Algo depois da função

Generators:


def meu_generator():
    for i in range(10):
        yield i

for valor in meu_generator():
    print(valor)

Context Managers:


class MeuContexto:
    def __enter__(self):
        print("Entrando no contexto")
        return self

    def __exit__(self, exc_type, exc_val, exc_tb):
        print("Saindo do contexto")

with MeuContexto() as contexto:
    print("Dentro do contexto")
# Saída:
# Entrando no contexto
# Dentro do contexto
# Saindo do contexto
